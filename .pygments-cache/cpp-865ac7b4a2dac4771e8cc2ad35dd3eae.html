int main(int, char**)
{
    VideoCapture cap(0); 
    if(!cap.isOpened()) 
        return -1;

    Mat frame;
    namedWindow("frame",1);

    while(1)
    {
        cap >> frame; // get a new frame from camera
        imshow("frame",frame);
        if(waitKey(30) >= 0) break;
    }

    return 0;
}
